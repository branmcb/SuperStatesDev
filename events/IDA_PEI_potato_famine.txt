# Potato Famine!
country_event = {
	id = potato_famine.1
	title = "potato_famine.1t"
	desc = "potato_famine.1d"
	picture = FAMINE_eventPicture
	
	fire_only_once = yes
	trigger = {
		OR = {
			tag = IDA
			tag = PEI
		}
		is_year = 1600
		NOT = { is_year = 1785 }
	}
	
	mean_time_to_happen = {
		months = 600
	
		modifier = {
			factor = 0.2
			NOT = { stability = -2 }
		}
		modifier = {
			factor = 0.5
			NOT = { stability = -1 }
		}
		modifier = {
			factor = 0.9
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 2
			stability = 1
		}
		modifier = {
			factor = 5
			stability = 2
		}
		modifier = {
			factor = 10
			stability = 3
		}
	}
	immediate = {
		set_country_flag = potato_famine
	}
	option = {
		name = "potato_famine.1opta"
		ai_chance = { factor = 50 }
		every_owned_province = {
			limit = {
				NOT = { is_capital = yes }
			}
			spawn_rebels = {
				type = particularist_rebels
				size = 1
			}
		}
	}
	option = {
		name = "potato_famine.1optb"
		ai_chance = { factor = 50 }
		add_years_of_income = -0.50
		add_adm_power = 10
	}
}

			}

					type = revolutionary_rebels
					size = 2
				}
			}
		}
	}
