#Serum Run Event
country_event = {
	id = flavorALA.1
	title = "flavorALA.1t"
	desc = "flavorALA.1d"
	picture = PLAUGE_eventPicture
	
	trigger = {
		tag = ALA
		NOT = { is_year = 1950 }
		num_of_cities = 3
	}
	
	mean_time_to_happen = {
		years = 100
	
		modifier = {
			factor = 1000
			OR = {
				is_month = 2	#March
				is_month = 3	#April
				is_month = 4	#May
				is_month = 5	#June
				is_month = 6	#July
				is_month = 7	#August
				is_month = 8	#September
				is_month = 9	#October
			}
		}
		modifier = {
			factor = 0.75
			OR = {
				is_month = 0	#January
				is_month = 1	#February
				is_month = 10	#November
				is_month = 11	#December
			}
		}
	}
	option = {
		name = "flavorALA.1opta"
		add_years_of_income = -0.125
		random_owned_province = {
			limit = {
				NOT = { is_capital = yes }
			}
			add_province_modifier = {
				name = "medicine_run"
				duration = 45
			}
		}
	}
	option = {
		name = "flavorALA.1optb"
		add_stability = -2
		add_prestige = -20
	}
}

#Discovery of Resources
country_event = {
	id = flavorALA.2
	title = "flavorALA.2t"
	desc = "flavorALA.2d"
	picture = EXPLORERS_eventPicture
	
	trigger = {
		tag = ALA
		num_of_cities = 2
		NOT = { is_year = 2000 }
		all_owned_province = {
			NOT = { development = 15 }
		}
	}
	
	mean_time_to_happen = {
		days = 36500
	
		modifier = {
			factor = 2
			had_country_flag = {
				flag = resource_discover_country
				days = 1825
			}
		}modifier = {
			factor = 2
			had_country_flag = {
				flag = resource_discover_country
				days = 3650
			}
		}
		modifier = {
			factor = 2
			had_country_flag = {
				flag = resource_discover_country
				days = 7300
			}
		}
		modifier = {
			factor = 0.90
			owns = 2618		#Yukon
		}
		modifier = {
			factor = 0.90
			owns = 2619		#Northwest Territories
		}
		modifier = {
			factor = 0.90
			owns = 2621		#Keewatin
		}
		modifier = {
			factor = 0.5
			alaska_area = {
				type = all
				owned_by = ROOT
			}
		}
		modifier = {
			factor = 5
			num_of_cities = 2
		}
		modifier = {
			factor = 0.90
			num_of_cities = 5
		}
		modifier = {
			factor = 0.90
			num_of_cities = 10
		}
		modifier = {
			factor = 1.25
			total_development = 25
		}
		modifier = {
			factor = 1.5
			total_development = 50
		}
		modifier = {
			factor = 2
			total_development = 100
		}
	}
	immediate = {
		random_core_province = {
			limit = {
				NOT = { is_capital = yes }
				NOT = { development = 15 }
			}
			hidden_effect = {
				set_province_flag = resource_discovered
			}
		}
		hidden_effect = {
			set_country_flag = resource_discover_country
		}
	}
	option = {
		name = "flavorALA.2opta"
		if = {
			limit = {
				any_core_province = { 
					NOT = { base_tax = 5 }
					has_province_flag = resource_discovered
				}
			}
			every_core_province = {
				limit = {
					NOT = { base_tax = 5 }
					has_province_flag = resource_discovered
				}
				add_base_tax = 1
				hidden_effect = {
					clr_province_flag = resource_discovered
				}
			}
			hidden_effect = {
				clr_country_flag = resource_discover_country
			}
		}
	}
	option = {
		name = "flavorALA.2optb"
		if = {
			limit = {
				any_core_province = { 
					NOT = { base_tax = 5 }
					has_province_flag = resource_discovered
				}
		 	}
			every_core_province = {
				limit = {
					NOT = { base_production = 5 }
					has_province_flag = resource_discovered
				}
				add_base_production = 1
				hidden_effect = {
					clr_province_flag = resource_discovered
				}
			}
			hidden_effect = {
				clr_country_flag = resource_discover_country
			}
		}
	}
	option = {
		name = "flavorALA.2optc"
		if = {
			limit = {
				any_core_province = { 
					NOT = { base_tax = 5 }
					has_province_flag = resource_discovered
				}
		 	}
			every_core_province = {
				limit = {
					NOT = { base_manpower = 5 }
					has_province_flag = resource_discovered
				}
				add_base_manpower = 1
				hidden_effect = {
					clr_province_flag = resource_discovered
				}
			}
			hidden_effect = {
				clr_country_flag = resource_discover_country
			}
		}
	}
	option ={
		name = "flavorALA.2optd"
		hidden_effect = {
			clr_country_flag = resource_discover_country
		}
		add_country_modifier = {
			name = tax_them_before_they_leave
			duration = 913
		}
	}
}
